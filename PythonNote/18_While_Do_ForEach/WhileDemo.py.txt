print("1~5:")
i = 1               # 초기식 
while i <= 5:       # 조건식 
    print(i)        # 실행문 
    i = i + 1       # 증감식

print("1~10까지 짝수:")
i = 1                   # 초기식 
while i <= 10:          # 조건식 
    if i % 2 == 0:      # 필터링 
        print(i)        # 실행문 
    i = i + 1           # 증감식
